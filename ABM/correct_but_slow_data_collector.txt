    def blood_flow(self):
        agent_oxy = [agent.oxy for agent in self.schedule.agents if type(agent) is Endothelial and agent.pos in self.wound_coord]
        oxy_total = sum(agent_oxy)
        return oxy_total / (len(self.wound_coord))

    def Collagen(self):
        agent_coll = [agent.coll for agent in self.schedule.agents if type(agent) is Endothelial and agent.pos in self.wound_coord]
        coll_total = sum(agent_coll)
        return coll_total / (len(self.wound_coord))
